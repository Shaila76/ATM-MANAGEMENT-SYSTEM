    private void GetBalance() {
        String Query = "select * from Information where acc_num='" + MyAccNum + "'";
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");
            Statement stmt = con.createStatement();
            ResultSet rs1 = stmt.executeQuery(Query);
            if (rs1.next()) {
                OldBalance = rs1.getInt(10);
                bal.setText("" + OldBalance + "." + "rs");
                this.dispose();
            } else {
//                    JOptionPane.showMessageDialog(this, "Wrong Account Number Or PIN");
            }
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);
        }

    private void jButton1MouseClicked(java.awt.event.MouseEvent evt) {                                      
        if (OldBalance < 100) {
            JOptionPane.showMessageDialog(this, "Not Have Enough Balance !");
        } else {
            try {
                String Query = "Update Information set bal = ? where acc_num = ? ";
                Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");
                PreparedStatement ps = con.prepareStatement(Query);
                ps.setInt(1, OldBalance - 100);
                ps.setInt(2, MyAccNum);
                if (ps.executeUpdate() == 1) {
                    JOptionPane.showMessageDialog(this, "100rs Withdrawn Successfully.");
                } else {
                    JOptionPane.showMessageDialog(this, "Withdrawal Failed !");
                }
            } catch (Exception e) {
                JOptionPane.showMessageDialog(this, e);
            }
        }
    }
