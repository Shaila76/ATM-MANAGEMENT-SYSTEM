    private void Clear() {
        acc_num.setText("");
        _name.setText("");
        _faname.setText("");
//        dob.setDate();
        addre.setText("");
        _pin.setText(" ");
        edu.setSelectedIndex(-1);
        occu.setText(" ");
        phone.setText("");

    }
    private void submit_btnMouseClicked(java.awt.event.MouseEvent evt) {                                        
        if (acc_num.getText().isEmpty() || _name.getText().isEmpty() || _faname.getText().isEmpty() || addre.getText().isEmpty() || _pin.getText().isEmpty() || occu.getText().isEmpty() || phone.getText().isEmpty()) {
            JOptionPane.showMessageDialog(this, "Please Fill Out All The Information !");
        }
        if (dob.getDate() == null) {
            JOptionPane.showMessageDialog(this, "Please select a valid Date of Birth.");
//        return;
        }
        String PhoneNumber = phone.getText();
        if (!PhoneNumber.matches("^[7-9][0-9]{9}$")) {
            JOptionPane.showMessageDialog(this, "Please Enter Valid 10-digit Number !");
        } else {
            try {

                Class.forName("com.mysql.cj.jdbc.Driver");
                Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");
                PreparedStatement pst = con.prepareStatement("insert into Information values(?,?,?,?,?,?,?,?,?,?)");
                pst.setInt(1, Integer.valueOf(acc_num.getText()));
                pst.setString(2, _name.getText());
                pst.setString(3, _faname.getText());
                pst.setString(4, addre.getText());
                pst.setInt(5, Integer.valueOf(_pin.getText()));
                pst.setString(6, edu.getSelectedItem().toString());
                pst.setString(7, occu.getText());
                pst.setString(8, PhoneNumber);
                pst.setString(9, dob.getDate().toString());
                pst.setInt(10, 0);
                int row = pst.executeUpdate();
                JOptionPane.showMessageDialog(this, "Account saved");
                con.close();
                Clear();

            } catch (Exception e) {
                JOptionPane.showMessageDialog(this, e);
            }
        }
    }
