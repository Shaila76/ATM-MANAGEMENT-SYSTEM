    private void CountDep() { 
    try 
        (Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");

    private void WithdrawMoney() {
        try {
            CountDep();
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");
            PreparedStatement add = con.prepareStatement("insert into Transactions(acc_num, Type, TDate, bal) values(?,?,?,?)");
            add.setInt(1, MyAccNum);
            add.setString(2, "Withdraw");
            add.setDate(3, new java.sql.Date(System.currentTimeMillis()));
            add.setString(4, amount.getText());
            int row = add.executeUpdate();
            con.close();
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);
        }
    }
    private void withbtnMouseClicked(java.awt.event.MouseEvent evt) {                                     
        String input = amount.getText().trim();
        if (input.isEmpty() || Integer.parseInt(input) <= 0) {

            JOptionPane.showMessageDialog(this, "Please Enter valid amount !");
        } else if (OldBalance < Integer.valueOf(input)) {
            JOptionPane.showMessageDialog(this, "Not Have Enough Balance !");
        } else {
            try {
                String Query = "Update Information set bal = ? where acc_num = ? ";
                Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atm", "root", "root");
                PreparedStatement ps = con.prepareStatement(Query);
                ps.setInt(1, OldBalance - Integer.parseInt(input));
                ps.setInt(2, MyAccNum);
                if (ps.executeUpdate() == 1) {
                    WithdrawMoney();
                    JOptionPane.showMessageDialog(this, "Withdrawal Successful.");
                    GetBalance();
                } else {
                    JOptionPane.showMessageDialog(this, "Withdrawal Failed !");
                }
            } catch (Exception e) {
                JOptionPane.showMessageDialog(this, e);
            }
        }
    }
